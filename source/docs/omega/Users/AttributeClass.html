def self.callbacks(callbacks = nil)
  unless callbacks.nil?
    @callbacks = callbacks
    @callbacks.each { |k,v| @callbacks[k] = v.is_a?(Array) ? v : [v] }
  end
  @callbacks
end

def self.invoke_callbacks(event, attribute)
  @callbacks ||= {}
  @callbacks[event].each { |cb|
    cb.call attribute
  } if @callbacks.has_key?(event)
end
