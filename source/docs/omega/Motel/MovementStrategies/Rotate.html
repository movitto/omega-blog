def move(loc, elapsed_seconds)
  unless valid?
    ::RJR::Logger.warn \
      "rotate movement strategy (#{rot_to_s}) not valid, not proceeding with move"
    return
  end

  ::RJR::Logger.debug \
    "moving location #{loc.id} via rotate movement strategy #{rot_to_s}"

  rotate(loc, elapsed_seconds)
end
