def valid?
  @id.is_a?(String) && @id != "" &&

  @material_id.is_a?(String) &&
  @material_id =~ /[a-zA-Z0-9]*-[a-zA-Z0-9]*/ &&

  !@entity_id.nil? &&
  (@entity.nil? || @entity.valid?) &&
   @quantity.numeric?
end
