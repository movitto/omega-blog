def initialize(sblock = nil, &block)
  unless sblock.nil?
    @sblock = sblock
    @block  = eval(sblock)
  end

  unless block.nil?
    @sblock = block.to_source
    @block  = block
  end

  if sblock.nil? && block.nil?
    raise RuntimeError, "must specify code block or string representation"
  end

  # TODO ensure block is a lambda/proc/other callable?
  super(@block)
end
