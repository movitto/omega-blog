Overview

Omega configuration manager, loads configuration options from config files, allowing the user to specify default values.

Config options can be accessed directly through instances of this class and consist of key (symbol) / value pairs.

def self.load(defaults = {})
  c = Config.new defaults

  CONFIG_FILES.each { |f|
    begin
      ff = File.expand_path(f)
      if File.file?(ff)
        c.update! YAML.load(File.open(ff))
      end
    rescue Exception => e
    end
  }

  c
end

def []=(key, value)
  if value.class == Hash
    @data[key.to_sym] = Config.new(value)
  else
    @data[key.to_sym] = value
  end
end

def has_attributes?(*required_attributes)
  # TODO multilevel / hash
  required_attributes.flatten.each { |attr|
    return false if !@data.has_key?(attr) || @data[attr].nil? || @data[attr] == ""
  }
  return true
end

def to_h
  cp = Hash[@data]
  cp.each { |k,v| cp[k] = v.to_h if v.is_a?(Config) }
  cp
end
