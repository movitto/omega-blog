def initialize(args = {})
  attr_from_args args, :timestamp => Time.now,
                       :handlers  =>       [],
                       :id        =>      nil,
                       :registry  =>      nil,
                       :type      =>      nil

  @timestamp = Time.parse(@timestamp) if @timestamp.is_a?(String)
end

def to_json(*a)
  {
    'json_class' => self.class.name,
    'data'       => json_data.merge({:handlers => handlers})
  }.to_json(*a)
end
