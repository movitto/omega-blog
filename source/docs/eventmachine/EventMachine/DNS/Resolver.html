def self.nameservers
  if !@nameservers
    @nameservers = []
    IO.readlines('/etc/resolv.conf').each do |line|
      if line =~ /^nameserver (.+)$/
        @nameservers << $1.split(/\s+/).first
      end
    end
  end
  @nameservers
end

def self.socket
  if !@socket || (@socket && @socket.error?)
    @socket = Socket.open

    @hosts  = {}
    IO.readlines('/etc/hosts').each do |line|
      next if line =~ /^#/
      addr, host = line.split(/\s+/)

      if @hosts[host]
        @hosts[host] << addr
      else
        @hosts[host] = [addr]
      end
    end
  end

  @socket
end
